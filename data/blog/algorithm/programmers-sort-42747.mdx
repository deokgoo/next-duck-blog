```javascript
function solution(citations) {
  let answer = 0;
  citations = citations.sort((a,b) => a>b?1:-1);
  for(let i=0;i<citations.length;i++) {
    if(citations[i]>=citations.length - i) {
      answer = citations.length - i;
      break;
    }
  }
  return answer;
}
```
